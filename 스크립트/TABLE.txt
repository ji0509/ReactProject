-- 사용자 즐겨찾기 메뉴 테이블
CREATE TABLE TB_SYS_FAVORITE
(
	USER_ID VARCHAR2(50) NOT NULL,
	PROGRAM_SEQ VARCHAR2(4) NOT NULL,
	SORT NUMBER(4,0),
	USE_YN VARCHAR2(1),
	INS_DT DATE,
	UPD_USER VARCHAR2(50),
	UPD_DT DATE,
	CONSTRAINTS TB_SYS_FAVORITE_PK PRIMARY KEY(USER_ID, PROGRAM_SEQ)
);

COMMENT ON TABLE TB_SYS_FAVORITE IS '즐겨찾기메뉴';

COMMENT ON COLUMN TB_SYS_FAVORITE.USER_ID IS '사용자ID';
COMMENT ON COLUMN TB_SYS_FAVORITE.PROGRAM_SEQ IS '메뉴코드';
COMMENT ON COLUMN TB_SYS_FAVORITE.SORT IS '정렬순서';
COMMENT ON COLUMN TB_SYS_FAVORITE.USE_YN IS '설정여부';
COMMENT ON COLUMN TB_SYS_FAVORITE.INS_DT IS '등록일자';
COMMENT ON COLUMN TB_SYS_FAVORITE.UPD_USER IS '수정자ID';
COMMENT ON COLUMN TB_SYS_FAVORITE.UPD_DT IS '수정일자';

CREATE INDEX TB_SYS_FAVORITE_IX01 ON TB_SYS_FAVORITE(USER_ID);

-- 공지사항 테이블
CREATE TABLE TB_SYS_NOTICE
(
	SEQNO NUMBER NOT NULL,
	VANGB VARCHAR2(4) NOT NULL,
	CONTENT VARCHAR2(4000) NOT NULL,
	INS_USER VARCHAR2(50),
	INS_DT DATE,
	UPD_USER VARCHAR2(50),
	UPD_DT DATE,
	CONSTRAINTS TB_SYS_NOTICE_PK PRIMARY KEY(SEQNO)
);

COMMENT ON TABLE TB_SYS_NOTICE IS '즐겨찾기메뉴';

COMMENT ON COLUMN TB_SYS_NOTICE.SEQNO IS '순번';
COMMENT ON COLUMN TB_SYS_NOTICE.VANGB IS '벤구분';
COMMENT ON COLUMN TB_SYS_NOTICE.CONTENT IS '내용';
COMMENT ON COLUMN TB_SYS_NOTICE.INS_USER IS '등록자ID';
COMMENT ON COLUMN TB_SYS_NOTICE.INS_DT IS '등록일자';
COMMENT ON COLUMN TB_SYS_NOTICE.UPD_USER IS '수정자ID';
COMMENT ON COLUMN TB_SYS_NOTICE.UPD_DT IS '수정일자';
